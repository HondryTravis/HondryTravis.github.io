var twoSum = function(nums, target) {
  var map = new Map()
  var ret = null
  for(var i=0, l = nums.length; i < l; i ++) {
    ret = target - nums[i]
    if(map.has(ret)) {
      return [map.get(ret), i]
    } else {
      map.set(nums[i], i)
    }
  }
  return []
};
